┌─ HOW IT WORKS ────────────────────────────────────────────────────────────┐
│                                                                             │
│ Example Usage:                                                              │
│                                                                             │
│   use scred_http::ConfigurableEngine;                                     │
│   use scred_redactor::RedactionEngine;                                    │
│   use std::sync::Arc;                                                      │
│                                                                             │
│   // Create engine                                                         │
│   let engine = Arc::new(RedactionEngine::new(Default::default()));       │
│   let config_engine = ConfigurableEngine::with_defaults(engine);          │
│                                                                             │
│   // Filtered detection (for logging)                                      │
│   let warnings = config_engine.detect_only("secret text");               │
│   // Only shows patterns matching detect_selector                         │
│                                                                             │
│   // Conservative redaction (for data safety)                             │
│   let redacted = config_engine.redact_only("secret text");               │
│   // Always redacts all patterns                                           │
│                                                                             │
│   // Combined operation                                                     │
│   let result = config_engine.detect_and_redact("secret text");           │
│   // Returns redacted text + filtered warnings                            │
│                                                                             │
