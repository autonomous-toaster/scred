PRODUCTION DEPLOYMENT GUIDE
━━━━━━━━━━━━━━━━━━━━━━━━

Quick Start - scred-cli:
  $ scred --help                    # See all options
  $ cat secrets.txt | scred         # Basic redaction
  $ SCRED_DEBUG=1 scred             # With debug logging
  
Quick Start - scred-proxy:
  $ SCRED_PROXY_UPSTREAM_URL=https://backend.example.com \
    ./scred-proxy
  # Now proxy runs on 0.0.0.0:9999

Quick Start - scred-mitm:
  $ ./scred-mitm &
  $ curl -x http://127.0.0.1:8080 https://api.example.com
  # Traffic gets redacted

Environment Variables:
  SCRED_DETECT_PATTERNS=CRITICAL,API_KEYS
  SCRED_REDACT_PATTERNS=CRITICAL,API_KEYS
  SCRED_PROXY_UPSTREAM_URL=https://backend.com
  SCRED_PROXY_LISTEN_PORT=9999
  SCRED_DEBUG=1                    # Enable debug logging
  SCRED_TRACE=1                    # Enable trace logging
